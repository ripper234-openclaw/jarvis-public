Analytics Tracking Code

Add this snippet before </body> on every page:

<!-- GoatCounter analytics -->
<script data-goatcounter="https://YOURCODE.goatcounter.com/count"
        async src="//gc.zgo.at/count.js"></script>

Replace YOURCODE with your GoatCounter site code.

Event Tracking (for funnels)

Add to pages that need funnel tracking:

// Track CTA clicks
document.querySelector('.cta-button')?.addEventListener('click', () => {
    window.goatcounter?.count({ path: '/events/cta-click', title: 'CTA Click', event: true });
});

// Track scroll depth
let tracked = {};
window.addEventListener('scroll', () => {
    const pct = Math.round(100 * window.scrollY / (document.body.scrollHeight - window.innerHeight));
    [25, 50, 80, 100].forEach(t => {
        if (pct >= t && !tracked[t]) {
            tracked[t] = true;
            window.goatcounter?.count({ path: '/events/scroll-' + t, title: 'Scroll ' + t + '%', event: true });
        }
    });
});
